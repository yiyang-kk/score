def a45_m_ratio(x):
    if np.abs((x - 0.450050))<=0.001:
        return 'x < 0.6567 or x is missing'
    else:
        return 'x >= 0.6567'
    
def a45_i_freq(x):
    if np.abs((x- 0.905324))<=0.001:
        return 'x is missing'
    else:
        return 'x is not missing'    
    
def a45_i_ratio(x):
    if np.abs((x -(-0.222326)))<=0.001:
        return 'x < 0.35855 or x is missing'
    else:
        return 'x >= 0.35855'    
    
def zh(x):
    if x==0:
        return 'x is missing'
    if np.abs(x - (-0.102122)) <=0.001:
        return 'x < 650'
    else:
        return 'x >= 650'
